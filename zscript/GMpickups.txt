class GMSwordBronze : CustomInventory // replaces GoldWandAmmo
{
    default
    {
        Inventory.Icon "UNKNA0";
        Inventory.Amount 1;
        Inventory.MaxAmount 5;
        Inventory.PickupMessage "Bronze Sword";
        +Inventory.Invbar
    }
    states
    {
    Spawn:
        UNKN A -1;
        Stop;
    Use:
        TNT1 A 0 {
            A_Log("Sword type is now Bronze.");
            GrandmasterPlayer(invoker.owner).GMSwordType = 1;
            GrandmasterPlayer(invoker.owner).GMSwordAttack = 5;
        }
        Fail;
    }
}

class GMPowerRing : CustomInventory replaces GoldWandAmmo // increases Strength
{
    default
    {
        Inventory.Icon "UNKNA0";
        Inventory.Amount 1;
        Inventory.MaxAmount 5;
        Inventory.PickupMessage "$GM_POWERRING";
        +Inventory.Invbar
    }
    states
    {
    Spawn:
        UNKN A -1;
        Stop;
    Use:
        TNT1 A 0 {
            A_Log("Strength increased.");
            GrandmasterPlayer(invoker.owner).GMStrength = (GrandmasterPlayer(invoker.owner).GMStrength + 1);
        }
        Stop;
    }
}
